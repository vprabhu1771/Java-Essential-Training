public class IfElseIfDemo {

    public static void main(String[] args) {
        int no = 0;

        if (no > 0)
        {
            System.out.println(no + " is positive no");
        }
        else if (no < 0)
        {
            System.out.println(no + " is Negative no");
        }
        else
        {
            System.out.println(no + " is zero");
        }
    }
}

Output

5 is positive no

-5 is Negative no

0 is zero
