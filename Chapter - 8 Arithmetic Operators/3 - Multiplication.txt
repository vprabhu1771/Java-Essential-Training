public class Multiplication {
    public static void main(String[] args) {
        int x = 10;

        int y = 2;

        System.out.print("x = ");

        System.out.println(x);

        System.out.print("y = ");

        System.out.println(y);

        System.out.println(x + " * " + y + " = " + (x*y));
    }
}

Output

x = 10

y = 2

10 * 2 = 20
