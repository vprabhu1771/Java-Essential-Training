public class Modulus {
    public static void main(String[] args) {
        int x = 7;

        int y = 3;

        System.out.print("x = ");

        System.out.println(x);

        System.out.print("y = ");

        System.out.println(y);

        System.out.println(x + " % " + y + " = " + (x%y));
    }
}

Output

x = 7

y = 3

7 % 3 = 1
