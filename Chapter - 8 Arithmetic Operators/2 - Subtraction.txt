public class Subtraction {
    public static void main(String[] args) {
        int x = 12;

        int y = 2;

        System.out.print("x = ");

        System.out.println(x);

        System.out.print("y = ");

        System.out.println(y);

        System.out.println(x + " - " + y + " = " + (x-y));
    }
}

Output

x = 12

y = 2

12 - 2 = 10
